CREATE TABLE flight (
    number CHAR(5),
    departureDate DATETIME, 
    departAirport VARCHAR(30), 
    arriveAirport VARCHAR(30), 
    arrivalDate DATETIME, 
    actualDepartureDate DATETIME, 
    actualArrivalDate DATETIME,
    airplaneId VARCHAR(10), 
    PRIMARY KEY (number, departureDate), 
    FOREIGN KEY (departAirport) REFERENCES airport (name) ON DELETE SET NULL ON UPDATE CASCADE, 
    FOREIGN KEY (arriveAirport) REFERENCES airport (name) ON DELETE SET NULL ON UPDATE CASCADE, 
    FOREIGN KEY (airplaneId) REFERENCES airplane (id) ON DELETE SET NULL ON UPDATE CASCADE
    );
